61void Listener::Subscribe(std::string name, std::function<bool(TEvent*)> callback, unsigned char priority) { // NOLINT(readability-function-size)
62 static_assert(std::is_base_of_v<IEvent, TEvent>, "Event T is not inherited from IEvent");
63 TOAST_INFO("Subscribing listener to event {0}", typeid(TEvent).name());
64
65 // Add function to callback map
66 std::pair callback_pair = { name, callback };
67 m_callbacks.emplace(typeid(TEvent), callback_pair);
68
69 // If the old priority is less than the new priority, we remove the old key to add it again with the new one 0x
70 if (m_events.contains(typeid(TEvent))) {
71 if (priority == 1) {
72 return; // We return if the priority is default to avoid checking more stuff 0x
73 }
74
75 for (auto it = TEvent::subscribers.begin(); it != TEvent::subscribers.end(); ++it) {
76 if (it->second == this && it->first < priority) {
77 TOAST_TRACE("Changing {0} priority from {1} to {2}", typeid(TEvent).name(), static_cast<int>(it->first), static_cast<int>(priority));
78 TEvent::subscribers.erase(it);
79 } else if (it->second == this && it->first >= priority) {
80 return;
81 }
82 }
83 }
84
85 // Add EventMap to event map
86 m_events.emplace(typeid(TEvent), &TEvent::subscribers);
87 // Add listeners to event subscribers map
88 TEvent::subscribers.emplace(priority, this);
89}

91template<typename TEvent>
92void Listener::Unsubscribe() {
93 static_assert(std::is_base_of_v<IEvent, TEvent>, "Event T is not inherited from IEvent");
94 size_t count = m_callbacks.count(typeid(TEvent));
95 if (!m_callbacks.contains(typeid(TEvent))) {
96 TOAST_WARN("Trying to unsubscribe from an event but there is no callback assigned, aborting...");
97 return;
98 }
99
100 // Remove functions from callback map
101 TOAST_INFO("Removing {0} callbacks from event {1}", count, typeid(TEvent).name());
102 m_callbacks.erase(typeid(TEvent));
103
104 // Remove EventMap* from event map
105 m_events.erase(typeid(TEvent));
106
107 // Remove listener from event subscribers map
108 for (auto it = TEvent::subscribers.begin(); it != TEvent::subscribers.end(); ++it) {
109 if (it->second == this) {
110 TEvent::subscribers.erase(it);
111 break;
112 }
113 }
114}

116template<typename TEvent>
117void Listener::Unsubscribe(std::string name) {
118 static_assert(std::is_base_of_v<IEvent, TEvent>, "Event T is not inherited from IEvent");
119
120 // CHECK if more functions exist
121 size_t count = m_callbacks.count(typeid(TEvent));
122 if (count == 0) {
123 TOAST_WARN("Trying to unsubscribe from an event but there is no callback assigned, aborting...");
124 return;
125 }
126 if (count == 1) {
127 TOAST_TRACE("Only one instance found, calling Unsubscribe from All");
128 Unsubscribe<TEvent>();
129 }
130
131 // Remove function from callback map
132 TOAST_INFO("Unsubscribing callback from event {0}", typeid(TEvent).name());
133 unsigned char deleted_functions = 0;
134 auto range = m_callbacks.equal_range(typeid(TEvent));
135 for (auto it = range.first; it != range.second;) {
136 // We any_cast it to our pair of string and function and then check if the names match
137 auto [callback_name, _] = it->second;
138 if (callback_name == name) {
139 it = m_callbacks.erase(it);
140 deleted_functions++;
141 } else {
142 // Need to check this to remove ALL functions that are the same
143 // not just the first one 0x
144 ++it;
145 }
146 }
147
148 // NOTE: Since a lot of you like to do typos while coding, i'll put a warning if no callbacks were unsubscribed 0x
149 if (deleted_functions == 0) { TOAST_WARN("No callbacks were unsubscribed, check \"{0}\" for typos", name); }

152template<typename TEvent>
153inline bool Listener::Dispatch(TEvent* event) {
154 // Find range with all callbacks
155 auto [begin, end] = m_callbacks.equal_range(typeid(TEvent));
156 bool handled = false;
157
158 // Call each callback
159 for (auto it = begin; it != end; ++it) {
160 auto [name, callback] = it->second;
161 handled = std::any_cast<std::function<bool(TEvent*)>>(callback)(event);
162 if (handled) { return true; }
163 }
164
165 return false;
166}
