95void InputSystem::CreateAction(const std::string& name, const json_t& data, int i) {
96 json_t action_data = data["actions"][i];
97
98 Action::Type action_type = Action::Button;
99 if (action_data["type"] == "button") {
100 action_type = Action::Button;
101 } else if (action_data["type"] == "axis1D") {
102 action_type = Action::Axis1D;
103 } else if (action_data["type"] == "axis2D") {
104 action_type = Action::Axis2D;
105 } else {
106 throw LayoutException(name, "Invalid action type");
107 }
108
109 int bind_count = data["actions"][i]["binds"].size();
110 if (!bind_count) { throw LayoutException(name, "A layout doesn't have any binds"); }
111
112 std::vector<Bind*> bind_vector(bind_count);
113 for (int j = 0; j < bind_count; j++) {
114 json_t bind_data = data["actions"][i]["binds"][j];
115 std::string type = bind_data["type"];
116 if (type == "keyboard") {
117 if (action_type != Action::Button) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
118
119 auto* bind = new PhysicalBind(Bind::Keyboard, bind_data["input"]);
120 bind_vector[j] = bind;
121
122 } else if (type == "mousePosition") {
123 if (action_type != Action::Axis2D) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
124
125 auto* bind = new PhysicalBind(Bind::MousePosition, 0);
126 bind_vector[j] = bind;
127
128 } else if (type == "mouseButton") {
129 if (action_type != Action::Button) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
130
131 auto* bind = new PhysicalBind(Bind::MouseButton, bind_data["input"]);
132 bind_vector[j] = bind;
133
134 } else if (type == "mouseScroll") {
135 if (action_type != Action::Axis1D) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
136
137 auto* bind = new PhysicalBind(Bind::MouseScroll, bind_data["input"]);
138 bind_vector[j] = bind;
139
140 } else if (type == "controllerButton") {
141 if (action_type != Action::Button) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
142
143 auto* bind = new PhysicalBind(Bind::ControllerButton, bind_data["input"]);
144 bind_vector[j] = bind;
145
146 } else if (type == "controllerTrigger") {
147 if (action_type != Action::Button) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
148
149 auto* bind = new PhysicalBind(Bind::ControllerTrigger, bind_data["input"]);
150 bind_vector[j] = bind;
151
152 } else if (type == "controllerStick") {
153 if (action_type != Action::Axis2D) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
154
155 auto* bind = new PhysicalBind(Bind::ControllerStick, bind_data["input"]);
156 bind_vector[j] = bind;
157
158 } else if (type == "composite1D") {
159 if (action_type != Action::Axis1D) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
160
161 auto* bind = new Composite1DBind(bind_data["input"][0], bind_data["input"][1]);
162 bind_vector[j] = bind;
163
164 } else if (type == "composite2D") {
165 if (action_type != Action::Axis2D) { throw ActionException(name, action_data["name"], "Wrong format for bind"); }
166
167 auto* bind = new Composite2DBind(bind_data["input"][0], bind_data["input"][1], bind_data["input"][2], bind_data["input"][3]);
168 bind_vector[j] = bind;
169
170 } else {
171 throw ActionException(name, data["actions"][i]["name"], "Bind type not found");
172 }
173 }
174
175 Action* action = new Action(data["actions"][i]["name"], action_type, bind_vector);
176
177 m_actions[action->name] = action;
178}

256void InputSystem::BindCheck(const std::string& action_name, Action* action, int key) { // NOLINT(readability-function-size)
257 for (Bind* bind : action->m_binds) {
258 switch (bind->type) {
259 case Bind::Keyboard: {
260 auto* phys = dynamic_cast<PhysicalBind*>(bind);
261 if (!phys || phys->keyId != key) { continue; }
262
263 int value = glm::clamp(static_cast<int>(m_keyData[key].state), 0, 1);
264
265 action->SetValue(Device::Keyboard, value, m_keyData[key], Time::uptime());
266 DispatchCallbacks(action_name, action);
267 break;
268 }
269
270 case Bind::Composite1D: {
271 auto* cmp = dynamic_cast<Composite1DBind*>(bind);
272 if (!cmp) { continue; }
273 if (cmp->left != key && cmp->right != key) { continue; }
274
275 int value = 0;
276 value -= glm::clamp(static_cast<int>(m_keyData[cmp->left].state), 0, 1);
277 value += glm::clamp(static_cast<int>(m_keyData[cmp->right].state), 0, 1);
278
279 action->SetValue(Device::Keyboard, value, m_keyData[key], Time::uptime());
280 DispatchCallbacks(action_name, action);
281 break;
282 }
283
284 case Bind::Composite2D: {
285 auto* cmp = dynamic_cast<Composite2DBind*>(bind);
286 if (!cmp) { continue; }
287 if (cmp->left != key && cmp->right != key && cmp->top != key && cmp->bottom != key) { continue; }
288
289 glm::vec2 value = {};
290 value.x -= glm::clamp(static_cast<int>(m_keyData[cmp->left].state), 0, 1);
291 value.x += glm::clamp(static_cast<int>(m_keyData[cmp->right].state), 0, 1);
292 value.y += glm::clamp(static_cast<int>(m_keyData[cmp->top].state), 0, 1);
293 value.y -= glm::clamp(static_cast<int>(m_keyData[cmp->bottom].state), 0, 1);
294
295 action->SetValue(Device::Keyboard, value, m_keyData[key], Time::uptime());
296 DispatchCallbacks(action_name, action);
297 break;
298 }
299
300 default: break;
301 }
302 }

378bool InputSystem::Controller(event::WindowInputDevice* event) {
379 if (event->event == GLFW_CONNECTED && glfwJoystickIsGamepad(event->jid)) {
380 m_controllers[event->jid] = GamepadState {};
381 TOAST_INFO("Controller {0} connected: {1}", event->jid, glfwGetGamepadName(event->jid));
382 return true;
383 }
384

386 TOAST_INFO("Controller {0} disconnected", event->jid);
387 m_controllers.erase(event->jid);
388 return true;
389 }
390
391 return false;
392}

394// NOTE: This won't work for multiple controller support, it will work for now as we don't need that feature
395// To begin implementing that, just pass the joystick ID to the functions
396void InputSystem::PollControllers() {
397 for (auto& [jid, state] : m_controllers) {
398 state.previous = state.current;
399 glfwGetGamepadState(jid, &state.current);
400 for (int i = 0; i <= GLFW_GAMEPAD_BUTTON_LAST; ++i) {
401 bool was_pressed = state.previous.buttons[i] == GLFW_PRESS;
402 bool is_pressed = state.current.buttons[i] == GLFW_PRESS;
403 if (!was_pressed && is_pressed) {
404 // Pressed
405 ControllerButton(i, Pressed);
406 } else if (was_pressed && is_pressed) {
407 // Held
408 ControllerButton(i, Hold);
409 } else if (was_pressed && !is_pressed) {
410 // Released
411 ControllerButton(i, Released);
412 }
413 }
414
415 for (int i = 0; i <= GLFW_GAMEPAD_AXIS_LAST; ++i) {
416 float prev = state.previous.axes[i];
417 float curr = state.current.axes[i];
418 if (std::abs(curr - prev) > 0.01f) {
419 std::array<float, 6> axes {};
420 std::copy(std::begin(state.current.axes), std::end(state.current.axes), axes.begin());
421 ControllerAxis(i, axes);
422 }
423 }
424 }
425}

443void InputSystem::ControllerAxis(int axis, std::array<float, 6> data) {
444 switch (axis) {
445 case GLFW_GAMEPAD_AXIS_LEFT_X:
446 case GLFW_GAMEPAD_AXIS_LEFT_Y: ControllerStick(0, glm::vec2 { data[0], -data[1] }); break;
447 case GLFW_GAMEPAD_AXIS_RIGHT_X:
448 case GLFW_GAMEPAD_AXIS_RIGHT_Y: ControllerStick(1, glm::vec2 { data[2], -data[3] }); break;
449 case GLFW_GAMEPAD_AXIS_LEFT_TRIGGER: ControllerTrigger(0, data[4]); break;
450 case GLFW_GAMEPAD_AXIS_RIGHT_TRIGGER: ControllerTrigger(1, data[5]); break;
